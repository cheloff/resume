% Fix margins so they don't suck balls
\usepackage[top=.5in, bottom=.5in, left=.75in, right=.75in]{geometry}

% Let's use a Helvetica for the entire document
\usepackage{helvet}
\renewcommand{\familydefault}{\sfdefault}

% Light grey for font color
\usepackage{xcolor}
\color{black!70}

% Link color will be red
\usepackage{hyperref}
\hypersetup{
	urlcolor=red,
	colorlinks=red,
	citecolor=red,
	colorlinks=true
}

% Make list spacing a little nicer
\setlength{\itemsep}{0pt}

% Change the default bullet point to a square
\renewcommand{\labelitemi}{$\blacksquare$}

% Style headers with titlesec
\usepackage{titlesec}

% h1
\titleformat*{\section}{\Huge\bfseries\color{red}}
\titlespacing*{\section}{0pt}{0pt}{0pt}

% h2
\titleformat*{\subsection}{\huge\bfseries}
\titlespacing*{\subsection}{0pt}{0pt}{10pt}

% h3
\titleformat*{\subsubsection}{\large\bfseries}
\titlespacing*{\subsubsection}{0pt}{15pt}{0pt}

%h4
\titlespacing*{\paragraph}{0pt}{5pt}{0pt}

% h5
\titleformat*{\subparagraph}{\small}
\titlespacing*{\subparagraph}{0pt}{0pt}{0pt}

% No page numbers
\pagenumbering{gobble}
